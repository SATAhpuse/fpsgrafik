function getFPS() { // biar smooth, fps berubah pelan pelan let change = (Math.random() - 0.5) * 8; currentFPS += change; if (currentFPS < 40) currentFPS = 40; if (currentFPS > 60) currentFPS = 60; return currentFPS; } /* Fungsi garis curve halus */ function drawSmoothGraph() { ctx.clearRect(0, 0, canvas.width, canvas.height); // Gradient fill let gradient = ctx.createLinearGradient(0, 0, 0, canvas.height); gradient.addColorStop(0, "rgba(190, 0, 255, 0.75)"); gradient.addColorStop(1, "rgba(60, 0, 100, 0.95)"); // Mulai path ctx.beginPath(); ctx.lineWidth = 3; ctx.strokeStyle = "#f2d6ff"; ctx.shadowBlur = 18; ctx.shadowColor = "#b100ff"; for (let i = 0; i < fpsData.length - 1; i++) { let x1 = (i / maxPoints) * canvas.width; let y1 = canvas.height - (fpsData[i] / 100) * canvas.height; let x2 = ((i + 1) / maxPoints) * canvas.width; let y2 = canvas.height - (fpsData[i + 1] / 100) * canvas.height; // midpoint untuk curve let cx = (x1 + x2) / 2; let cy = (y1 + y2) / 2; if (i === 0) ctx.moveTo(x1, y1); // Quadratic curve smooth ctx.quadraticCurveTo(x1, y1, cx, cy); } ctx.stroke(); ctx.shadowBlur = 0; // Fill bawah grafik ctx.lineTo(canvas.width, canvas.height); ctx.lineTo(0, canvas.height); ctx.closePath(); ctx.fillStyle = gradient; ctx.fill(); } /* Update Real-Time */ function updateFPS() { let fps = getFPS(); document.getElementById("fpsValue").innerText = fps.toFixed(0); fpsData.push(fps); if (fpsData.length > maxPoints) fpsData.shift(); drawSmoothGraph(); } // Isi awal biar penuh for (let i = 0; i < maxPoints; i++) { fpsData.push(getFPS()); } // Animasi lebih smooth (60fps) function animate() { updateFPS(); requestAnimationFrame(animate); } animate();
